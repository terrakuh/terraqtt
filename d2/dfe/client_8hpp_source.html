259  std::size_t process_one(std::error_code& ec,
260  std::size_t available = std::numeric_limits<std::size_t>::max())
261  {
262  _read_context.available = available;
263 
264  _skip_ignored();
265  if (!_read_context.available) {
266  return 0;
267  } // read new packet
268  else if (_read_type == protocol::Control_packet_type::reserved) {
269  _read_type = protocol::peek_type(*_input, ec, _read_context);
270  if (ec) {
271  return 0;
272  }
273  }
274 
275  // process type
276  switch (_read_type) {
277  case protocol::Control_packet_type::reserved: break;
278  case protocol::Control_packet_type::connack: _handle_connack(ec); break;
279  case protocol::Control_packet_type::publish: _handle_publish(ec); break;
280  case protocol::Control_packet_type::puback: _handle_puback(ec); break;
281  case protocol::Control_packet_type::pubrec: _handle_pubrec(ec); break;
282  case protocol::Control_packet_type::pubrel: _handle_pubrel(ec); break;
283  case protocol::Control_packet_type::pubcomp: _handle_pubcomp(ec); break;
284  case protocol::Control_packet_type::suback: _handle_suback(ec); break;
285  case protocol::Control_packet_type::unsuback: _handle_unsuback(ec); break;
286  case protocol::Control_packet_type::pingresp: _handle_pingresp(ec); break;
287  default: ec = Error::bad_packet_type; break;
288  }
289  if (ec) {
290  TERRAQTT_LOG(ERROR, "Error during packet handling: {}", ec.message());
291  }
292 
293  _skip_ignored();
294  const std::size_t processed = available - _read_context.available + _overread;
295  _overread = 0;
296  TERRAQTT_LOG(TRACE, "Processed {} bytes", processed);
297  return processed;
298  }
